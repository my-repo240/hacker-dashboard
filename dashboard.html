let tasks = []; function openModal() { document.getElementById("modal").style.display = "flex"; } function closeModal() { document.getElementById("modal").style.display = "none"; } function addTask() { const name = document.getElementById("taskName").value.trim(); const start = document.getElementById("taskStart").value; const due = document.getElementById("taskDue").value; const status = document.getElementById("taskStatus").value; const priority = document.getElementById("taskPriority").value; if (!name || !start || !due) { alert("Please fill in all fields."); return; } const task = { id: Date.now(), name, start, due, status, priority, }; tasks.push(task); renderTasks(); updateTaskCounts(); closeModal(); } function renderTasks() { const table = document.getElementById("taskTable"); table.innerHTML = tasks .map( (task, index) => ` ${index + 1} ${task.name} ${task.status} ${task.start} ${task.due} ${task.priority} ` ) .join(""); } function updateTaskCounts() { const counts = { new: 0, progress: 0, testing: 0, complete: 0, }; tasks.forEach((task) => { counts[task.status]++; }); document.getElementById("count-new").textContent = counts.new; document.getElementById("count-progress").textContent = counts.progress; document.getElementById("count-testing").textContent = counts.testing; document.getElementById("count-complete").textContent = counts.complete; }
